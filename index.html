    <h2>今日经文</h2>

    <div id="verse">正在加载经文...</div>


    <script>

        // 纪元日（计算的起始点），格式：年, 月-1, 日

        const EPOCH_DATE = new Date(Date.UTC(2024, 0, 1)); // 2024年1月1日

        const MS_PER_DAY = 1000 * 60 * 60 * 24;


        // 这是一个假设的经文数组。在实际部署时，你应该从服务器加载你的 verses.txt

        // 为了演示，我们先使用硬编码的数组。

        const VERSES = [

            "创世记 1:1 | 起初，神创造天地。",

            "诗篇 23:1 | 耶和华是我的牧者，我必不至缺乏。",

            "约翰福音 3:16 | 神爱世人，甚至将他的独生子赐给他们，叫一切信他的，不至灭亡，反得永生。",

            "腓立比书 4:13 | 我靠着那加给我力量的，凡事都能做。",

            // ... 放入您所有的经文 ...

        ];


        function getDailyVerse() {

            const today = new Date();

            // 设置今天的 UTC 时间为午夜，以确保每日计算的准确性

            const today_utc = Date.UTC(today.getFullYear(), today.getMonth(), today.getDate());


            // 计算从纪元日到今天的总天数

            const days_diff = Math.floor((today_utc - EPOCH_DATE) / MS_PER_DAY);


            // 通过取模运算确保索引在经文总数范围内，实现循环

            const verse_index = days_diff % VERSES.length;


            // 显示经文

            document.getElementById('verse').innerText = VERSES[verse_index];

        }


        // 立即运行

        getDailyVerse();

    </script>
